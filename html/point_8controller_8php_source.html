515  if(!$mode || !in_array($mode,$mode_arr)) $mode = 'update';

527  // Change points
528  $args = new stdClass();

530  $args->point = $current_point;
531 
532  switch($mode)
533  {
534  case 'add' :
535  $args->point += $point;
536  break;
537  case 'minus' :
538  $args->point -= $point;
539  break;
540  case 'update' :
541  case 'signup' :
542  $args->point = $point;
543  break;
544  }
545  if($args->point < 0) $args->point = 0;
546  $point = $args->point;
547

549  $trigger_obj = new stdClass();
550  $trigger_obj->member_srl = $args->member_srl;
551  $trigger_obj->mode = $mode;
552  $trigger_obj->current_point = $current_point;
553  $trigger_obj->current_level = $current_level;
554  $trigger_obj->set_point = $point;

556  if(!$trigger_output->toBool())
557  {
558  return $trigger_output;
559  }
560 
561  // begin transaction
562  $oDB = &DB::getInstance();
563  $oDB->begin();
564 
565  // If there are points, update, if no, insert
566  $oPointModel = getModel('point');
567  if($oPointModel->isExistsPoint($member_srl)) executeQuery("point.updatePoint", $args);
568  else executeQuery("point.insertPoint", $args);
569 
570  // Get a new level
571  $level = $oPointModel->getLevel($point, $config->level_step);
572 
573  // If existing level and a new one are different attempt to set a point group
574  if($level != $current_level)
575  {
576  // Check if the level, for which the current points are prepared, is calculate and set the correct group
577  $point_group = $config->point_group;
578  // If the point group exists
579  if($point_group && is_array($point_group) && count($point_group) )
580  {
581  // Get the default group
582  $default_group = $oMemberModel->getDefaultGroup();
583  // Get the removed group and the newly granted group
584  $del_group_list = array();
585  $new_group_list = array();
586 
587  asort($point_group);
588  // Reset group after initialization
589  if($config->group_reset != 'N')
590  {
591  // If the new level is in the right group
592  if(in_array($level, $point_group))
593  {
594  // Delete all groups except the one which the current level belongs to
595  foreach($point_group as $group_srl => $target_level)
596  {
597  $del_group_list[] = $group_srl;
598  if($target_level == $level) $new_group_list[] = $group_srl;
599  }
600  }
601  // Otherwise, in case the level is reduced, add the recent group
602  else
603  {
604  $i = $level;
605  while($i > 0)
606  {
607  if(in_array($i, $point_group))
608  {
609  foreach($point_group as $group_srl => $target_level)
610  {
611  if($target_level == $i)
612  {
613  $new_group_list[] = $group_srl;
614  }
615  }
616  $i = 0;
617  }
618  $i--;
619  }
620  }
621  // Delete the group of a level which is higher than the current level
622  foreach($point_group as $group_srl => $target_level)
623  {
624  if($target_level > $level) $del_group_list[] = $group_srl;
625  }
626  $del_group_list[] = $default_group->group_srl;
627  }
628  // Grant a new group
629  else
630  {
631  // Check until the current level by rotating setting the configurations of the point groups
632  foreach($point_group as $group_srl => $target_level)
633  {
634  $del_group_list[] = $group_srl;
635  if($target_level <= $level) $new_group_list[] = $group_srl;
636  }
637  }
638  // If there is no a new group, granted the default group
639  if(!$new_group_list[0]) $new_group_list[0] = $default_group->group_srl;
640  // Remove linkage group
641  if($del_group_list && count($del_group_list))
642  {
643  $del_group_args = new stdClass;
644  $del_group_args->member_srl = $member_srl;
645  $del_group_args->group_srl = implode(',', $del_group_list);
646  $del_group_output = executeQuery('point.deleteMemberGroup', $del_group_args);
647  }
648  // Grant a new group
649  foreach($new_group_list as $group_srl)
650  {
651  $new_group_args = new stdClass;
652  $new_group_args->member_srl = $member_srl;
653  $new_group_args->group_srl = $group_srl;

655  }
